## Usage

```sh
const times = [
    "2018-12-19",
    "2019-02-19",
    "2019-10-01",
    "2020-10-01",
    "2020-10-10",
    "2020-11-10",
    "2021-08-08",
    "2021-10-16",
    "2021-12-08",
    "2021-12-16",
    "2022-01-11",
    "2022-09-09",
    "2022-09-11",
    "2022-11-09",
  ];
  
const App = () => {
    return (
        <>
            <TimeLineF data = {times}/>
        </>
    );
}

OptionalProps/InitSizeDefault

initTLSize = {
    width: 800,  //containersize
    height: 120,  //containersize
    arrowsWidth: 10, //sliderwidth
    pointWidth: 2, //timepintwidth
    top: "2%", //position
    left: "5%", //position 
}

```
